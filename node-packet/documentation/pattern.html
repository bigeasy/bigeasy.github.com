number = (pattern, index) ->
  if match = /^(&)(\d+)(.*)$/.exec pattern
    [ !! match[1], parseInt(match[2], 10), index + match[1].length + match[2].length, match[2] ]
  else if match = /^(&)0x([0-9a-f]+)(.*)$/i.exec pattern
